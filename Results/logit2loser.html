getdeciles<-function(vname){
    deciles<-matrix(NA,nrow=nr,ncol=10,dimnames=list(NULL,c("D1","D2","D3","D4","D5","D6","D7","D8","D9","D10")))
    r<-1
    c<-1
    for (i in 1:dim(y.test.mat)[1]){
        deciles[r,c]<-y.test.mat[i,vname]
        r<-r+1
        if (r>nr){
            r<-1
            c<-c+1
        }
    }
    cat(vname)
    print(summary(deciles))
    boxplot(deciles,main=paste(vname," Distribution of by Predicted Decile"),
            ylab=paste(vname," Return"),xlab="Decile",col="blue")    
}
